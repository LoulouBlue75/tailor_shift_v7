### 5.2 Scoring Algorithm

```typescript
function calculateAssessmentScore(
  answers: Answer[],
  roleLevel: string,
  division?: string
): AssessmentResult {
  
  // Get role-specific weights
  const weights = ROLE_WEIGHTS[roleLevel];
  
  // Calculate raw dimension scores
  const rawScores = {};
  for (const dim of DIMENSIONS) {
    const dimAnswers = answers.filter(a => a.dimension === dim);
    rawScores[dim] = average(dimAnswers.map(a => a.score));
  }
  
  // Apply role-level weighting
  let totalScore = 0;
  for (const dim of DIMENSIONS) {
    totalScore += rawScores[dim] * weights[dim];
  }
  
  // Determine level
  const level = 
    totalScore >= 81 ? 'expert' :
    totalScore >= 61 ? 'advanced' :
    totalScore >= 41 ? 'proficient' :
    'developing';
  
  return {
    scores: rawScores,
    totalScore,
    level
  };
}
```

1. Introduce harder situational scenarios
2. Add behavioral questions requiring depth
3. Validate consistency across sub-capabilities
